checks
  select * from smc_keys where type = 'fpe2';
  select * from smc_keys where type in ('sp5a', 'sp78', 'fpe2');
  ./smc -l | grep 'fpe2'

sp78: (val[0] * 256 + val[1]) / 256.0
fpe2: (int(val[0]) << 6) + (int(val[1]) >> 2)
sp5a: (val[0] * 256 + val[1]) / 1024.0
