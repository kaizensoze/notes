build from source
  git clone https://github.com/facebook/osquery.git
  cd osquery
  make deps
  make

test
  make test

binaries
  ./build/darwin/osquery/

run
  ./build/darwin/osquery/osqueryi --verbose

  ./smc -l | grep 'fpe2'
