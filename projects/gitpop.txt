get flask working with nginx
github login

setup
  sudo apt-get install python-pip python-dev nginx
  sudo pip install virtualenv
  cd ~/gitpop
  virtualenv gitpop
  source gitpop/bin/activate
  pip install uwsgi https://github.com/mitsuhiko/flask/tarball/master PyGithub
  flask --app=app initdb
  flask --app=app --host=0.0.0.0 run

  import flask
